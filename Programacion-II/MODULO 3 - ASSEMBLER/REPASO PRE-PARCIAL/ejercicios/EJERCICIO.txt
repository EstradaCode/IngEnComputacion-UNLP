ORG 1000H
N DB 1,20,5,4,30
MODIFICADOS DB 0
ORG 3000H
SUBRUTINA: PUSH BX
PUSH AX
MOV BX, SP
ADD BX, 6
MOV BX,[BX]; TENGO LA DIRECCION DE N
MOV AL,0; RES 0
MOV AH,1 ; I
BUCLE: ADD AL,AH
CMP AH, [BX]; COMPARO I CON N
JZ FIN
INC AH
JMP BUCLE
FIN: MOV [BX],AL
POP AX
POP BX
RET


ORG 2000H
MOV BX, OFFSET N
MOV AL, OFFSET MODIFICADOS - OFFSET N
LOOOP: PUSH BX
MOV BL,[BX]
CMP BL, 20
JNS SIGO
CALL SUBRUTINA
INC MODIFICADOS
SIGO: POP BX
INC BX
DEC AL
JNZ LOOOP
FINAL: HLT
END
