ORG 3000H
CALC: ADD DL, AL
ADD DL, AH
SUB DL, CL
RET

org 2000h
mov al, A
mov AH, B
MOV CL, C
CALL CAL
MOV D, DL
HLT 
END


B- POR VALOR PILA


CALC: MOV BX, SP 
ADD BX, 2 | PARA MOVERSE DE LA DIR DE RET
MOV DL, BYTE PTR [BX] -> CASTEA EL DATO A PARA QUE ENTRE EN UN BYTE
ADD BX, 2
MOV AH, BYTE PTR[BX] | DATO B
ADD BX, 2
MOV CL, BYTE PTR[BX] | DATO C
ADD DL, AH -> SUMAN
SUB DL, CL ->RESTA
RET






MOV AX, C
PUSH AX
MOV AX, B |
PUSH AX 
MOV AX, A |
PUSH AX
CALL CALC
MOV D, DL
POP AX
POP AX
POP AX
HLT 
END


C- POR REF PILA
SUBRT: MOV BX, SP 
ADD BX, 2
PUSH BX // GUARDO EL DATO
MOV BX, [BX]
MOV DL, [BX] // DATO A EN AL
POP BX // RECUPERO DIRECCION 
ADD BX, 2
PUSH BX 
MOV BX, [BX]
MOV AH, [BX]
POP BX
ADD BX, 2
MOV BX, [BX]
MOV CL, [BX] // DATO C EN CL
ADD DL, AH ; SUMAN 
SUB DL, CL ; RESTAN 
RET




MOV AX, OFFSET C 
PUSH AX 
MOV AX, OFFSET B
PUSH AX
MOV AX, OFFSET A
PUSH AX 
CALL SUBRT
MOV D, DL 
POP AX
POP AX 
POP AX
HLT
END
