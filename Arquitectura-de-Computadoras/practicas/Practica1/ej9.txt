// ROTARIZQ

ORG 1000H
NUM1 DB 128

ORG 3000H
ROTARIZQ: PUSH BX
PUSH AX
MOV Al,[BX]
ADD AL, AL
ADC AL,0
FIN: MOV [BX],AL
POP AX
POP BX
RET

ORG 2000H
; LETRA POR VALOR 
; CADENA POR REF TODO PILA
MOV BX, OFFSET NUM1
CALL ROTARIZQ
HLT 
END


// ROTARIZQ_N Y ROTARDER_N

ORG 1000H
NUM1 DB 3
ROT DB 2

ORG 3000H
ROTARDER_N:PUSH BX; SE OBTIENE ROTANDO 8-N POSICIONES
PUSH AX
PUSH CX
MOV CL,8
SUB CL, AH
MOV AH,CL
CALL ROTARIZQ_N
POP CX
POP AX
POP BX
RET

ROTARIZQ_N: PUSH BX
PUSH AX
MOV Al,[BX]
CMP AH,1; VERIFICAR QUE SEA ROT POSITIVA
JS FIN
CMP AH, 0 ; VERIFICAR QUE NO SEA 0
JZ FIN
DO: ADD AL, AL
ADC AL,0
DEC AH
JNZ DO
FIN: MOV [BX],AL
POP AX
POP BX
RET

ORG 2000H
; LETRA POR VALOR 
; CADENA POR REF TODO PILA
MOV BX, OFFSET NUM1
MOV AH, ROT
CALL ROTARDER_N
HLT 
END
